import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import {
  AlertFilledGeneral,
  AlertFilledError,
  AlertFilledWarning,
  AlertFilledInfo,
  AlertFilledSuccess,
} from './index';

<Title>Alert Variants</Title>
<Subtitle>
  Two additional variants are available – outlined, and filled:
</Subtitle>

### Import

```jsx
import Alert from 'tokenomt-ui/feedback/Alert';
import CheckFilledIcon from 'tokenomy-ui/data-display/Icon/CheckFilledIcon';
import AlertIcon from 'tokenomy-ui/data-display/Icon/AlertIcon';
import CloseFilledIcon from 'tokenomy-ui/data-display/Icon/CloseFilledIcon';
import InfoIcon from 'tokenomy-ui/data-display/Icon/InfoIcon';
```

### Usage

<AlertFilledGeneral />

```jsx
<Alert className="MuiAlert-standardFilledGeneral" {...props}>
  This is a general alert — check it out!
</Alert>
```

<AlertFilledError />

```jsx
<Alert variant="filled" severity="error" icon={<CloseFilledIcon />} {...props}>
  This is an error alert — check it out!
</Alert>
```

<AlertFilledWarning />

```jsx
<Alert variant="filled" severity="warning" icon={<AlertIcon />} {...props}>
  This is a warning alert — check it out!
</Alert>
```

<AlertFilledInfo />

```jsx
<Alert variant="filled" severity="info" icon={<InfoIcon />} {...props}>
  This is an info alert — check it out!
</Alert>
```

<AlertFilledSuccess />

```jsx
<Alert
  variant="filled"
  severity="success"
  icon={<CheckFilledIcon />}
  {...props}
>
  This is a success alert — check it out!
</Alert>
```
