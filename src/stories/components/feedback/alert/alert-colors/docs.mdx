import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import {
  AlertColorGeneral,
  AlertColorError,
  AlertColorWarning,
  AlertColorInfo,
  AlertColorSuccess,
} from './index';

<Title>Alert Colors</Title>
<Subtitle>
  The <code>color</code> prop will override the default color for the specified
  severity.
</Subtitle>

### Import

```jsx
import Alert from 'tokenomy-ui/feedback/Alert';
import CheckFilledIcon from 'tokenomy-ui/data-display/Icon/CheckFilledIcon';
import AlertIcon from 'tokenomy-ui/data-display/Icon/AlertIcon';
import CloseFilledIcon from 'tokenomy-ui/data-display/Icon/CloseFilledIcon';
import InfoIcon from 'tokenomy-ui/data-display/Icon/InfoIcon';
```

### Usage

<AlertColorGeneral />

```jsx
<Alert className="MuiAlert-standardGeneral" color="error" {...props}>
  This is a general alert — check it out!
</Alert>
```

<AlertColorError />

```jsx
<Alert severity="error" color="error" icon={<CloseFilledIcon />} {...props}>
  This is a error alert — check it out!
</Alert>
```

<AlertColorWarning />

```jsx
<Alert severity="warning" color="warning" icon={<AlertIcon />} {...props}>
  This is a warning alert — check it out!
</Alert>
```

<AlertColorInfo />

```jsx
<Alert severity="info" color="info" icon={<InfoIcon />} {...props}>
  This is a info alert — check it out!
</Alert>
```

<AlertColorSuccess />

```jsx
<Alert severity="success" color="success" icon={<CheckFilledIcon />} {...props}>
  This is a success alert — check it out!
</Alert>
```
