import {
  AlertColorGeneral,
  AlertColorError,

} from './index';

<Title>Alert Colors</Title>
<Subtitle>
  The <code>color</code> prop will override the default color for the specified
  severity.
</Subtitle>

### Import

```jsx
import Alert from 'tokenomy-ui/lab/Alert';
```

### Usage

<AlertGeneral />

```jsx
<Alert className="MuiAlert-standardGeneral" color="error">
  This is a general alert — check it out!
</Alert>
```

<AlertError />

```jsx
<Alert severity="error" color="error">
  This is a error alert — check it out!
</Alert>
```

<AlertWarning />

```jsx
<Alert severity="warning" color="warning">
  This is a warning alert — check it out!
</Alert>
```

<AlertInfo />

```jsx
<Alert severity="info" color="info">
  This is a info alert — check it out!
</Alert>
```

<AlertSuccess />

```jsx
<Alert severity="success" color="success">
  This is a success alert — check it out!
</Alert>
```
