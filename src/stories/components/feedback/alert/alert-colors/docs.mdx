import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import {
  AlertColorGeneral,
  AlertColorError,
  AlertColorWarning,
  AlertColorInfo,
  AlertColorSuccess,
} from './index';

<Title>Alert Colors</Title>
<Subtitle>
  The <code>color</code> prop will override the default color for the specified
  severity.
</Subtitle>

### Import

```jsx
import React from 'react';
import Alert from 'tokenomy-ui/feedback/Alert';
import CheckFilledIcon from '@components/data-display/Icon/CheckFilledIcon';
import AlertIcon from '@components/data-display/Icon/AlertIcon';
import CloseFilledIcon from '@components/data-display/Icon/CloseFilledIcon';
import InfoIcon from '@components/data-display/Icon/InfoIcon';
```

### Usage

<AlertColorGeneral />

```jsx
export const AlertColorGeneral = (props) => (
  <Alert className="MuiAlert-standardGeneral" color="error" {...props}>
    This is a general alert — check it out!
  </Alert>
);
```

<AlertColorError />

```jsx
export const AlertColorError = (props) => (
  <Alert severity="error" color="error" icon={<CloseFilledIcon />} {...props}>
    This is a error alert — check it out!
  </Alert>
);
```

<AlertColorWarning />

```jsx
export const AlertColorWarning = (props) => (
  <Alert severity="warning" color="warning" icon={<AlertIcon />} {...props}>
    This is a warning alert — check it out!
  </Alert>
);
```

<AlertColorInfo />

```jsx
export const AlertColorInfo = (props) => (
  <Alert severity="info" color="info" icon={<InfoIcon />} {...props}>
    This is a info alert — check it out!
  </Alert>
);
```

<AlertColorSuccess />

```jsx
export const AlertColorSuccess = (props) => (
  <Alert
    severity="success"
    color="success"
    icon={<CheckFilledIcon />}
    {...props}
  >
    This is a success alert — check it out!
  </Alert>
);
```
