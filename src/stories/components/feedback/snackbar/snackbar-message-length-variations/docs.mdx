<Title>Snackbar message length variations</Title>
<Subtitle>Some snackbars with varying message length.</Subtitle>

### Import

```jsx
import Snackbar from 'tokenomy-ui/feedback/Snackbar';
import Button from 'tokenomy-ui/inputs/Button';
```

### Usage

<SnackbarMessageShorWithButton />

```jsx
export const SnackbarMessageShorWithButton = (props) => {
  const action = (
    <Button variant="contained" color="primary">
      lorem ipsum dolorem
    </Button>
  );

  return (
    <SnackbarContent message="I love snacks." action={action} {...props} />
  );
};
```

<SnackbarMessageLong />

```jsx
export const SnackbarMessageLong = (props) => {
  return (
    <SnackbarContent
      message={
        'I love candy. I love cookies. I love cupcakes. \
    I love cheesecake. I love chocolate.'
      }
      {...props}
    />
  );
};
```
