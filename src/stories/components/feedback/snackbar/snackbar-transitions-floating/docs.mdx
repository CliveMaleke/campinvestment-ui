import { SnackbarTransitionFloating } from './index';

<Title>Transition floating snackbars</Title>
<Subtitle>Snackbars should appear above FABs (on mobile).</Subtitle>

### Import

```jsx
import Button from 'tokenomy-ui/inputs/Button';
import Snackbar from 'tokenomy-ui/feedback/Snackbar';
import Snackbar from 'tokenomy-ui/feedback/Snackbar';
import Alert from 'tokenomy-ui/feedback/Alert';
import CloseIcon from 'tokenomy-ui/data-display/Icon/CloseIcon';
```

### Usage

```jsx
const useStyles = makeStyles((theme) =>
  createStyles({
    '@global': {
      body: {
        backgroundColor: theme.palette.background.paper,
      },
    },
    menuButton: {
      marginRight: theme.spacing(2),
    },
    fab: {
      position: 'absolute',
      bottom: theme.spacing(2),
      right: theme.spacing(2),
    },
    snackbar: {
      [theme.breakpoints.down('xs')]: {
        bottom: 90,
      },
    },
  }),
);

export const SnackbarTransitionFloating = (props) => {
  const classes = useStyles();

  return (
    <React.Fragment>
      <CssBaseline />
      <div>
        <AppBar position="static" color="primary">
          <Toolbar>
            <Typography variant="h6" color="inherit">
              App Bar
            </Typography>
          </Toolbar>
        </AppBar>
        <Fab color="secondary" className={classes.fab}>
          +
        </Fab>
        <Snackbar
          open
          autoHideDuration={6000}
          message="Archived"
          action={<button color="inherit">Undo</button>}
          className={classes.snackbar}
          {...props}
        >
          <Alert
            severity="info"
            className="MuiAlert-snackbar MuiAlert-standardGeneral"
            action={
              <span aria-label="close" color="inherit" onClick={() => {}}>
                <CloseIcon fontSize="small" />
              </span>
            }
          >
            I love snacks!
            <div className="MuiButtonWrapper-root">
              <Button variant="text" onClick={() => {}}>
                Button 1
              </Button>
              <Button variant="text" onClick={() => {}}>
                Button 2
              </Button>
            </div>
          </Alert>
        </Snackbar>
      </div>
    </React.Fragment>
  );
};
```
