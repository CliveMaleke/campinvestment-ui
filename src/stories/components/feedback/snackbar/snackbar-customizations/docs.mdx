### Import

```jsx
import React from 'react';
import Button from 'tokenomy-ui/inputs/Button';
import AlertTitle from 'tokenomy-ui/lab/AlertTitle';
import Snackbar from 'tokenomy-ui/feedback/Snackbar';
import Alert from 'tokenomy-ui/feedback/Alert';
import CloseIcon from 'tokenomy-ui/data-display/Icon/CloseIcon';
import CheckFilledIcon from 'tokenomy-ui/data-display/Icon/CheckFilledIcon';
import AlertIcon from 'tokenomy-ui/data-display/Icon/AlertIcon';
import CloseFilledIcon from 'tokenomy-ui/data-display/Icon/CloseFilledIcon';
import InfoIcon from 'tokenomy-ui/data-display/Icon/InfoIcon';
```

### Usage

<SnackbarCustomizedGeneral />

```jsx
export const SnackbarCustomizedGeneral = (props) => {
  const [open, setOpen] = React.useState(false);

  const handleClick = () => {
    setOpen(true);
  };

  const handleClose = (
    event: React.SyntheticEvent | React.MouseEvent,
    reason?: string,
  ) => {
    if (reason === 'clickaway') {
      return;
    }

    setOpen(false);
  };

  return (
    <>
      <Button variant="contained" color="primary" onClick={handleClick}>
        Open general snackbar
      </Button>
      <Snackbar
        open={open}
        autoHideDuration={1000}
        onClose={handleClose}
        {...props}
      >
        <Alert
          onClose={handleClose}
          className="MuiAlert-snackbar MuiAlert-standardGeneral"
          action={
            <span aria-label="close" color="inherit" onClick={handleClose}>
              <CloseIcon fontSize="small" />
            </span>
          }
        >
          <AlertTitle className="MuiAlert-snackbarTitle">
            This is a general message!
          </AlertTitle>
          Description
          <div className="MuiButtonWrapper-root">
            <Button variant="text" onClick={() => {}}>
              Button 1
            </Button>
            <Button variant="text" onClick={() => {}}>
              Button 2
            </Button>
          </div>
        </Alert>
      </Snackbar>
    </>
  );
};
```
