import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import GridSpacing from './index';

<Title>Spacing</Title>
<Subtitle>
  The responsive grid focuses on consistent spacing widths, rather than column
  width. Material Design margins and columns follow an 8px square baseline grid.
  The spacing property is an integer between 0 and 10 inclusive. By default, the
  spacing between two grid items follows a linear function:{' '}
  <code>output(spacing) = spacing * 8px</code>, e.g. <code>spacing={2}</code>{' '}
  creates a 16px wide gap.
</Subtitle>

### Import

```jsx
import Grid from 'tokenomy-ui/layout/Grid';
```

<h3 style={{ marginBottom: 25 }}>Usage</h3>

<GridSpacing />

```jsx
<Grid container spacing={2}>
  <Grid key={1} item>
    ...
  </Grid>
  <Grid key={2} item>
    ...
  </Grid>
  <Grid key={3} item>
    ...
  </Grid>
</Grid>
```
