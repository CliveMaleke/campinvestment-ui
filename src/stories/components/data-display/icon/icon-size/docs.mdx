import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import { IconSmall, IconMedium, IconLarge, IconCustomSize } from './index';

<Title>Size</Title>
<Subtitle>
  Icon color has a size variant that you can specify its use.
</Subtitle>

### Import

```jsx
import HomeIcon from 'tokenomy-ui/data-display/Icon/HomeIcon';
```

### Usage

### Small

<IconSmall />

```jsx
<HomeIcon fontSize="small" />
```

### Default / Medium

<IconMedium />

```jsx
<HomeIcon />
```

### Large

<IconLarge />

```jsx
<HomeIcon fontSize="large" />
```

### Custom

<IconCustomSize />

```jsx
<HomeIcon style={{ fontSize: 40 }} />
```
