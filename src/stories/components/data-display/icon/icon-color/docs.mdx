import { Title, Subtitle } from '@storybook/addon-docs/blocks';
import { IconPrimary, IconSecondary, IconInfo, IconSuccess, IconWarning, IconDanger, IconDisabled } from './index';

<Title>Color</Title>
<Subtitle>
  Icon color has a color variant that you can specify its use.
</Subtitle>

### Import

```jsx
import HomeIcon from '@components/data-display/Icon/HomeIcon';
```

### Usage

### Primary

<IconPrimary />

```jsx
<HomeIcon color="primary" /> 
```

### Secondary

<IconSecondary />

```jsx
<HomeIcon color="secondary" /> 
```

### Info

<IconInfo />

```jsx
<HomeIcon style={{ color: '#1EAFED' }} /> 
```

### Success

<IconSuccess />

```jsx
<HomeIcon style={{ color: '#16D091' }} />
```

### Warning

<IconWarning />

```jsx
<HomeIcon style={{ color: '#FCAC19' }} /> 
```

### Danger

<IconDanger />

```jsx
<HomeIcon style={{ color: '#FF5374' }} /> 
```

### Disabled

<IconDisabled />

```jsx
<HomeIcon color="disabled" /> 
```
